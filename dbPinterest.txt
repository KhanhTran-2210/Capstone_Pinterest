CREATE TABLE binh_luan(
	binh_luan_id INT PRIMARY KEY auto_increment,
	ngay_binh_luan DATE,
	noi_dung VARCHAR(255),
	nguoi_dung_id INT,
	FOREIGN KEY (nguoi_dung_id) REFERENCES nguoi_dung(nguoi_dung_id),
	hinh_id INT,
	FOREIGN KEY (hinh_id) REFERENCES hinh_anh(hinh_id)
)
INSERT INTO binh_luan (ngay_binh_luan, noi_dung, nguoi_dung_id, hinh_id)
SELECT
    DATE_ADD('2023-01-01', INTERVAL id DAY) AS ngay_binh_luan,
    CONCAT('Nội dung bình luận ', id) AS noi_dung,
    FLOOR(RAND() * 30) + 1 AS nguoi_dung_id,

    LIMIT 30
) AS numbers;
